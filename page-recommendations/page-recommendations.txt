# Write your MySQL query statement below
select distinct(L.page_id) as recommended_page
from(
    select user2_id as friend_id
    from Friendship
    where user1_id=1
    union 
    select user1_id as friend_id
    from Friendship
    where user2_id=1
) F
join Likes L
on F.friend_id=L.user_id
where L.page_id not in(
    select page_id 
    from Likes
    where user_id=1
)
